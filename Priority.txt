#include<stdio.h>
#include<conio.h>

int main()
{
    int n,Extime[50],BT[50],PO[50],CT[50],WT[50],W1=0,_TAT=0;
    int i,j,s,AT[10],TT[50];
    float AWT,ATAT;
    printf("Enter Number of processes= ");
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        printf("The process %d :\n",i+1);
        printf("Enter the Arival Time=",i+1);
        scanf("%d",&AT[i]);
        printf("Enter The Burst Time of Process= ");
        scanf("%d",&BT[i]);
        printf("Enter the priority of process%d= ",i+1);
        scanf("%d",&PO[i]);
        Extime[i]=i+1;
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            AT[0]>AT[i];
            if(PO[i]>PO[j])
            {
                s=PO[i];
                PO[i]=PO[j];
                PO[j]=s;
                s=BT[i];
                BT[i]=BT[j];
                BT[j]=s;
                s=Extime[i];
                Extime[i]=Extime[j];
                Extime[j]=s;
            }

        }
    }
    for(i=0;i<n;i++)
    {
        WT[0]=0;
        CT[0]=BT[0];
            TT[i]=CT[i]-AT[i];
            WT[i]=TT[i]-BT[i];
            CT[i+1]=CT[i]+BT[i+1];
            W1=W1+WT[i];
            _TAT=_TAT+TT[i];
            AWT =W1/n;
            ATAT = _TAT/n;
    }

    printf("\n process\tAT\tBT\tWT\tCOM\tTT\tPO \n");
    for(i=0;i<n;i++)
        printf("%d \t\t %d \t %d \t %d \t %d \t %d \t %d\n",Extime[i],AT[i],BT[i],WT[i],CT[i],TT[i],PO[i]);
        printf("AW=%f\t ATAT=%f \n",AWT,ATAT);
    return 0;
}
